    Create src/components/AudioPlayer.jsx with this exact structure:

    ```jsx
    import AudioPlayerLib from "react-h5-audio-player"
    import "react-h5-audio-player/lib/styles.css"
    import { useRef, useState, useCallback } from "react"

    const SPEEDS = [0.5, 1, 1.25, 1.5, 2]

    export function AudioPlayer({ moduleId, onComplete }) {
      const playerRef = useRef(null)
      const completedRef = useRef(false)
      const [speed, setSpeed] = useState(1)

      const applySpeed = useCallback((newSpeed) => {
        if (playerRef.current?.audio?.current) {
          playerRef.current.audio.current.playbackRate = newSpeed
        }
      }, [])

      const handleSpeedChange = useCallback((newSpeed) => {
        setSpeed(newSpeed)
        applySpeed(newSpeed)
      }, [applySpeed])

      // Re-apply speed on every play event (iOS Safari resets playbackRate on resume)
      const handlePlay = useCallback(() => {
        applySpeed(speed)
      }, [applySpeed, speed])

      // Auto-complete at 90% — fires once per session (completedRef guard)
      const handleListen = useCallback((e) => {
        const audio = e.target
        if (!completedRef.current && audio.duration > 0) {
          if (audio.currentTime / audio.duration >= 0.9) {
            completedRef.current = true
            onComplete()
          }
        }
      }, [onComplete])

      return (
        <div style={{ color: "var(--text-primary)" }}>
          <AudioPlayerLib
            ref={playerRef}
            src={`/audio/${moduleId}/overview.mp3`}
            onListen={handleListen}
            onPlay={handlePlay}
            listenInterval={1000}
            showJumpControls={false}
            showSkipControls={false}
            layout="stacked"
            style={{
              backgroundColor: "var(--bg-card)",
              color: "var(--text-primary)",
              borderRadius: "12px",
            }}
          />
          {/* Speed selector — custom UI because react-h5-audio-player has no playbackRate prop */}
          <div
            className="flex gap-2 justify-center pt-2 pb-1"
            role="group"
            aria-label="Playback speed"
          >
            {SPEEDS.map((s) => (
              <button
                key={s}
                onClick={() => handleSpeedChange(s)}
                className="text-xs px-2.5 py-1 rounded-lg cursor-pointer transition-colors"
                style={{
                  color: speed === s ? "var(--text-primary)" : "var(--text-muted)",
                  border: speed === s
                    ? "1px solid var(--border-accent)"
                    : "1px solid var(--border)",
                  backgroundColor: "var(--bg-card)",
                }}
              >
                {s}x
              </button>
            ))}
          </div>
        </div>
      )
    }
    ```
