**src/__tests__/storage.migration.test.js** (covers DATA-02):
```js
import { describe, it, expect, beforeEach, afterEach } from "vitest"

describe("loadProgress — mindmap → deck migration", () => {
  beforeEach(() => {
    localStorage.clear()
  })
  afterEach(() => {
    localStorage.clear()
  })

  it("migrates mindmap key to deck in stored completed data", () => {
    const stored = {
      completed: { "1": { audio: true, mindmap: true, flashcards: false, quiz: false, teachback: false } },
      notes: {},
      streak: { current: 2, best: 5, lastStudyDate: "2026-03-28" },
      celebratedMilestones: [],
    }
    localStorage.setItem("ai-tracker-progress", JSON.stringify(stored))

    // Import after setting localStorage so loadProgress reads it
    const { loadProgress } = await import("../utils/storage.js")
    const result = loadProgress()

    expect(result.completed["1"].deck).toBe(true)
    expect(result.completed["1"].mindmap).toBeUndefined()
    expect(result.streak.current).toBe(2)
  })

  it("migration is idempotent — running twice does not corrupt deck value", () => {
    const stored = {
      completed: { "1": { audio: false, mindmap: true, flashcards: false, quiz: false, teachback: false } },
      notes: {},
      streak: { current: 0, best: 0, lastStudyDate: null },
      celebratedMilestones: [],
    }
    localStorage.setItem("ai-tracker-progress", JSON.stringify(stored))

    const { loadProgress, saveProgress } = await import("../utils/storage.js")
    const first = loadProgress()
    saveProgress(first)
    const second = loadProgress()

    expect(second.completed["1"].deck).toBe(true)
    expect(second.completed["1"].mindmap).toBeUndefined()
  })

  it("does not touch data that already uses deck key", () => {
    const stored = {
      completed: { "1": { audio: false, deck: true, flashcards: false, quiz: false, teachback: false } },
      notes: {},
      streak: { current: 0, best: 0, lastStudyDate: null },
      celebratedMilestones: [],
    }
    localStorage.setItem("ai-tracker-progress", JSON.stringify(stored))

    const { loadProgress } = await import("../utils/storage.js")
    const result = loadProgress()

    expect(result.completed["1"].deck).toBe(true)
    expect(result.completed["1"].mindmap).toBeUndefined()
  })
})
```
